class Spins:

    """
    Los objetos de esta clase corresponden a grillas cuadradas con espines hacia arriba o hacia abajo en cada uno de sus puntos.

    Attributes:
        lattice (list): Lista de listas que define la grilla de la configuraci√≥n.
        size (int): Tama√±o de la grilla cuadrada.

    Methods:
        calculate_energy(self): Calcula la energ√≠a de la grilla.
        calculate_magnetization(self): Calcula la magnetizaci√≥n de la grilla.
        update(self, i, j, T, rand): Actualiza la grilla con base en el algoritmo de Metr√≥polis.

    """

    def __init__(self, init_state, size):
        """Constructor de las grillas de espines. Cuenta con los atributos lattice (arreglo correspondiente a la grilla) y size (tama√±o de la grilla).

        Examples:

        Se crea una grilla de tama√±o 2 con espines hacia abajo y se obtiene su tama√±o.

            >>> spins_t = Spins(-1, 2)
            >>> spins_t.size
            2

        Args:
            init_state (int o float): Valor que indica el estado inicial de la grilla (-1: espines hacia abajo, 0: espines aleatorios, cualquier otro: espines hacia arriba).
            size (int): Tama√±o de la grilla cuadrada.

        """

        if init_state == 0:
            self.lattice = np.random.choice([-1, 1], size=(size, size))
        elif init_state == -1:
            self.lattice = np.full((size, size), -1)
        else:
            self.lattice = np.ones((size, size))
        self.size = size

    def calculate_energy(self):

    	"""C√°lculo de la energƒ±ÃÅa respecto a los vecinos m√°s cercanos en la grilla.

        Examples:

        Se crea una grilla de tama√±o 2 con espines hacia abajo y se calcula su energ√≠a.

	    >>> spins_t = Spins(-1, 2)
            >>> spins_t.calculate_energy()
	    -8.0

	Returns:
	    (float): Suma de interacci√≥n con los cuatro espines m√°s cercanos por el par√°metro energ√©tico de interacci√≥n -J.

	"""

    	neighbors_sum = (np.roll(self.lattice, 1, axis=1)
    	    + np.roll(self.lattice, -1, axis=1) + np.roll(self.lattice, 1, axis=0) + np.roll(self.lattice, -1, axis=0)) 

    	return -0.5 * (self.lattice * neighbors_sum).sum()

    def calculate_magnetization(self):

        """C√°lculo de la magnetizaci√≥n seg√∫n los espines de una grilla.
        Examples:

        Se crea una grilla de tama√±o 2 y se calcula su magnetizaci√≥n.

            >>> spins_t = Spins(-1, 2)
            >>> spins_t.calculate_magnetization()
            -4

        Returns:
            (int): Magnetizaci√≥n de la configuraci√≥n.
        """    

        return np.sum(self.lattice)       

    def update(self, i, j, T, rand):
        """C√°lculo del cambio de energƒ±ÃÅa entre los espines m√°s cercanos en la grilla y c√°lculo del cambio de la magnetizaci√≥        n con base en el algoritmo de metr√≥polis.

        Examples:

        Se crea una grilla de tama√±o 4 con espines hacia abajo y se actualiza el spin en la posici√≥n (2, 2) bajo una tempera        tura de 2 y una probabilidad aleatoriqa -0.1.

	    >>>	spin_t = Spins(-1, 4)
            >>> spin_t.update(2, 2, 2, 1)
            (0, 0)

	Args:		
            i (int): Fila (posici√≥n en x).
			j (int): Columna (posici√≥n en y).
            T (int or float): Temperatura.
            rand (int or float): N√∫mero aleatorio (pseudo-aleatorio) con el que se ejecuta el algoritmo de metr√≥polis. Este             √∫ltimo contiene comparaciones del argumento con el logaritmo natural de la probabilidad dada por la distribuci√≥n de Boltzmann, por lo que generalmente se usan valores negativos.

	Returns:
		dE (int): Si el valor aleatorio dado es menor a la probabilidad dada por la distribuci√≥n de Boltzmann, da el valor del cambio de la energƒ±ÃÅa respecto a los espines m√°s cercanos. Si no, retorna 0.
 		dM (int): Si el valor aleatorio dado es menor a la probabilidad dada por la distribuci√≥n de Boltzmann, da el valor del cambio de la magnetizaci√≥n respecto a los espines m√°s cercanos. Si no, retorna 0.

	"""

        dE = (
        2
        * self.lattice[i, j]
        * (
            self.lattice[(i + 1) % self.size, j]
            + self.lattice[(i - 1) % self.size, j]
            + self.lattice[i, (j + 1) % self.size]
            + self.lattice[i, (j - 1) % self.size]
        )
        )
        if rand < - dE / (T):
            self.lattice[i, j] *= -1
            dM = 2 * self.lattice[i, j]
            return dE, dM

        return 0, 0

def simulate_Ising(Ti, Tf, steps, size, mcsteps, thermsteps, init_state=1, J=1, kb=1):
    """Ejecuci√≥n del modelo de Ising para un intervalo de temperaturas especificado a partir de una configuraci√≥n inicial y la generaci√≥n de n√∫meros psuedo-aleatorios (m√©todo de Monte Carlo). Primero se lleva a cabo la termalizaci√≥n y luego se guardan los valores seg√∫n los steps de Montecarlo.

    Examples:

    Se somete una grilla de tama√±o 3 con espines hacia arriba al modelo de Ising con 5 steps de Montecarlo, 2 steps de terma    lizaci√≥n y un intervalo de temperaturas que va de 0.5 a 5 con 3 steps. El resultado var√≠a con la semilla, la cual es tomada de la hora del sistema.

        >>> simulate_Ising(Ti=0.5, Tf=5, steps=3, size=3, mcsteps=5, thermsteps=2, init_state=1)
        (array([[-18., -18., -18., -18., -18.], [-18., -18., -18., -18., -18.], [-18., -18., -18., -18., -18.]]),
 		array([[9., 9., 9., 9., 9.], [9., 9., 9., 9., 9.], [9., 9., 9., 9., 9.]]),
 		array([0.5 , 2.75, 5.  ]))

    Args:
        Ti (int or float): Temperatura inicial.
        Tf (int or float): Temperatura final.
        steps (int): Cantidad de temperaturas en el intervalo indicado.
        size (int): Tama√±o de la grilla cuadrada.
        mcsteps (int): Pasos del m√©todo de Montecarlo.
        thermsteps (int): Pasos de termalizaci√≥n.
        init_state (int or float): Valor con el que se inicializa la grilla de espines seg√∫n el constructor.
        J (int or float, optional): Propiedad magn√©tica del material (fijado como 1 por defecto). 
        kb(int or float, optional): Constante de Boltzmann (fijado como 1 por defecto).

    Returns: 
        energies (list): Lista de listas con las energ√≠as de la configuraci√≥n para cada temperatura.
        magnetizations (list): Lista con las magnetizaciones de la configuraci√≥n para cada temperatura.
        temperatures (list): Lista con las temperaturas empleadas.

    """

    spins = Spins(init_state, size)
    E = spins.calculate_energy()
    M = spins.calculate_magnetization()

    temperatures = np.linspace(Ti, Tf, steps)
    energies = np.zeros((steps, mcsteps))
    magnetizations = np.zeros((steps, mcsteps))

    rand_pos = np.random.randint(size, size=((steps, mcsteps + thermsteps, 2)))
    rands = np.log(np.random.uniform(size=(steps, mcsteps + thermsteps)))

    for k in range(steps):

        for l in range(mcsteps + thermsteps):
            i, j = rand_pos[k, l]

            dE, dM = spins.update(i, j, temperatures[k], rands[k, l])
            E += dE
            M += dM

            if l >= thermsteps:
                energies[k, l-thermsteps] = E
                magnetizations[k, l-thermsteps] = M


    return energies, magnetizations, temperatures
